async function driveRequest(url, options={}) {

  const res = await fetch(url, {

    ...options,

    headers: {

      'Authorization': 'Bearer ' + gAccessToken,

      ...(options.headers||{})

    }

  });

  if(!res.ok) {

    const text = await res.text();

    throw new Error(`Drive API ${res.status}: ${text}`);

  }

  return res.json();

}


async function findOrCreateFolder(name, parentId) {

  const q = `name='${name}' and mimeType='application/vnd.google-apps.folder' and '${parentId}' in parents and trashed=false`;

  const result = await driveRequest(`https://www.googleapis.com/drive/v3/files?q=${encodeURIComponent(q)}&fields=files(id,name)`);

  if(result.files && result.files.length > 0) return result.files[0].id;

  const created = await driveRequest('https://www.googleapis.com/drive/v3/files', {

    method: 'POST',

    headers: {'Content-Type': 'application/json'},

    body: JSON.stringify({ name, mimeType: 'application/vnd.google-apps.folder', parents: [parentId] })

  });

  return created.id;

}

async function saveToDrive() {

  if(!gAccessToken) return;

  const keys = ['props','txns','certs','appliances','maint','owners','rentHistory'];

  const data = {};

  keys.forEach(k => { data[k] = state[k]; });

  try {

    if(gDriveDataFileId) {

        method: 'PATCH',

        headers: { 'Authorization': 'Bearer ' + gAccessToken, 'Content-Type': 'application/json' },

        body: blob

      });

    } else {

      const form = new FormData();

      form.append('metadata', new Blob([JSON.stringify({

        name: DATA_FILENAME,

        parents: [gDriveFolderIds.root || 'root']

      })], {type:'application/json'}));

      form.append('file', blob);

      const res = await fetch('https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart&fields=id', {

        method: 'POST',

        headers: { 'Authorization': 'Bearer ' + gAccessToken },

        body: form

      gDriveDataFileId = created.id;

    }

  } catch(e) {

    console.error('Drive save error:', e);

    showSync('⚠ Save failed', true);

  }

}


async function uploadFileToDrive(base64Data, mimeType, filename, propName, category) {

  if(!gAccessToken) return null;

  try {

    if(!gDriveFolderIds.root) {

      gDriveFolderIds.root = await findOrCreateFolder(DRIVE_FOLDER_NAME, 'root');

    }

    const propKey = propName || 'General';

    if(!gDriveFolderIds[propKey]) {

      gDriveFolderIds[propKey] = await findOrCreateFolder(propKey, gDriveFolderIds.root);

    }

    const catKey = propKey + '/' + (category||'Documents');

    if(!gDriveFolderIds[catKey]) {

      gDriveFolderIds[catKey] = await findOrCreateFolder(category||'Documents', gDriveFolderIds[propKey]);

    }

    const byteChars = atob(base64Data);

    const bytes = new Uint8Array(byteChars.length);

    for(let i=0;i<byteChars.length;i++) bytes[i]=byteChars.charCodeAt(i);

    const blob = new Blob([bytes], {type: mimeType});

    const form = new FormData();

    form.append('metadata', new Blob([JSON.stringify({

      name: filename,

      parents: [gDriveFolderIds[catKey]]

    })], {type:'application/json'}));

    form.append('file', blob);

    const res = await fetch('https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart&fields=id,webViewLink', {

      method: 'POST',

      headers: { 'Authorization': 'Bearer ' + gAccessToken },

      body: form

    });

    const created = await res.json();

    showSync('✓ Saved to Drive');

    return created.webViewLink;

  } catch(e) {

    console.error('Drive upload error:', e);

    showSync('⚠ Upload failed', true);

    return null;

  }

}

function uid() { return Date.now().toString(36) + Math.random().toString(36).slice(2,5); }

function fmtMoney(n) { return '£' + Math.abs(n).toLocaleString('en-GB',{minimumFractionDigits:0,maximumFractionDigits:0}); }

function fmtDate(d) { if(!d||d==='—') return '—'; const p=d.split('-'); return p[2]+'/'+p[1]+'/'+p[0]; }

function daysUntil(d) { return Math.round((new Date(d)-new Date())/(1000*60*60*24)); }

function certStatus(expiry) {

  if(!expiry) return 'grey';

  const d = daysUntil(expiry);

  if(d < 0) return 'red';

  if(d < 60) return 'amber';

  return 'green';

}

function certLabel(expiry) {

  if(!expiry) return 'No date';

  const d = daysUntil(expiry);

  if(d < 0) return 'Expired';

  if(d < 60) return d+'d left';

  return 'Valid';

}

function getOwnerShare(prop) {

  if(ownerEntry) return ownerEntry.share/100;

function getPeriodRange(period) {

  const now = new Date();

  const y = now.getFullYear(), m = now.getMonth(), d = now.getDate();

  let start, end, prevStart, prevEnd, label;

  if(period==='ytd') {

    start = new Date(y,0,1); end = new Date(y,m,d,23,59,59);

    prevStart = new Date(y-1,0,1); prevEnd = new Date(y-1,m,d,23,59,59);

    label = 'YTD '+y;

  } else if(period==='last12') {

    start = new Date(y-1,m,d+1); end = new Date(y,m,d,23,59,59);

    prevStart = new Date(y-2,m,d+1); prevEnd = new Date(y-1,m,d,23,59,59);

    label = 'Last 12 months';

  } else if(period==='q1') {

    start = new Date(y,0,1); end = new Date(y,2,31,23,59,59);

    prevStart = new Date(y-1,0,1); prevEnd = new Date(y-1,2,31,23,59,59);

    label = 'Q1 '+y;

  } else if(period==='q2') {

    start = new Date(y,3,1); end = new Date(y,5,30,23,59,59);

    prevStart = new Date(y-1,3,1); prevEnd = new Date(y-1,5,30,23,59,59);

    label = 'Q2 '+y;

  } else if(period==='q3') {

    start = new Date(y,6,1); end = new Date(y,8,30,23,59,59);

    prevStart = new Date(y-1,6,1); prevEnd = new Date(y-1,8,30,23,59,59);

    label = 'Q3 '+y;

  } else if(period==='q4') {

    start = new Date(y,9,1); end = new Date(y,11,31,23,59,59);

    prevStart = new Date(y-1,9,1); prevEnd = new Date(y-1,11,31,23,59,59);

    label = 'Q4 '+y;

  } else if(period==='tytd') {

    const taxStart = now >= new Date(y,3,6) ? new Date(y,3,6) : new Date(y-1,3,6);

    start = taxStart; end = new Date(y,m,d,23,59,59);

    prevStart = new Date(taxStart.getFullYear()-1,3,6);

    prevEnd = new Date(taxStart.getFullYear()-1, m, d, 23,59,59);

    const taxY = taxStart.getFullYear();

    label = `Tax YTD ${taxY}/${String(taxY+1).slice(2)}`;

  }

  return {start,end,prevStart,prevEnd,label};

}


function txnsInRange(start, end) {

  const fpIds = state.filterProp ? [state.filterProp] : state.props.map(p=>p.id);

  return state.txns.filter(t=>{

    if(!t.date) return false;

    const d = new Date(t.date);

    return d>=start && d<=end && fpIds.includes(t.prop);

  });

}

function sumIncome(txns) { return txns.filter(t=>t.type==='income').reduce((s,t)=>s+txnShare(t),0); }

function sumExpense(txns) { return txns.filter(t=>t.type==='expense').reduce((s,t)=>s+txnShare(t),0); }


function deltaChip(curr, prev) {

  if(prev===0) return '';

  const diff = curr - prev;

  const pct = ((diff/prev)*100).toFixed(1);

  const up = diff >= 0;

  const color = up ? 'var(--green)' : 'var(--red)';

  const arrow = up ? '▲' : '▼';

  return `<span style="font-size:11px;color:${color}">${arrow} ${fmtMoney(Math.abs(diff))} (${Math.abs(pct)}%)</span>`;

}

function getTaxYear(date) {

  const d = new Date(date);

  const y = d.getFullYear();

  const cutoff = new Date(y, 3, 6); // 6 April

  return d >= cutoff ? y : y - 1;

}

function getQuarter(date) {

  const d = new Date(date);

  const y = getTaxYear(date);

  const q1s = new Date(y, 3, 6), q1e = new Date(y, 6, 5);

  const q2s = new Date(y, 6, 6), q2e = new Date(y, 9, 5);

  const q3s = new Date(y, 9, 6), q3e = new Date(y+1, 0, 5);

  const q4s = new Date(y+1, 0, 6), q4e = new Date(y+1, 3, 5);

  if(d>=q1s&&d<=q1e) return {q:1,label:'Q1',period:'6 Apr – 5 Jul',due:'7 Aug'};

  if(d>=q2s&&d<=q2e) return {q:2,label:'Q2',period:'6 Jul – 5 Oct',due:'7 Nov'};

  if(d>=q3s&&d<=q3e) return {q:3,label:'Q3',period:'6 Oct – 5 Jan',due:'7 Feb'};

  if(d>=q4s&&d<=q4e) return {q:4,label:'Q4',period:'6 Jan – 5 Apr',due:'7 May'};

  return {q:0,label:'',period:'',due:''};

}

function buildCSV(txns) {

function propFirstWord(propId) {

  const p = getProp(propId);

  const fullName = p ? p.name : 'Unknown';

  const words = fullName.replace(/[^a-zA-Z0-9 ]/g,'').trim().split(/\s+/);

  return words.find(w => isNaN(w)) || words[0];

}


function buildDocFilename(propId, date, type, suffix) {

  const roadName = propFirstWord(propId);

  const d = date ? new Date(date) : new Date();

  const yymm = String(d.getFullYear()).slice(2) + String(d.getMonth()+1).padStart(2,'0');

  const detail = type.split(/[\s&\-_]+/).filter(Boolean)

    .map((w,i) => i===0 ? w.charAt(0).toUpperCase()+w.slice(1) : w.charAt(0).toUpperCase()+w.slice(1))

    .join('').replace(/[^a-zA-Z0-9]/g,'');

  const ext = _lastUploadedMime==='application/pdf' ? '.pdf' : (_lastUploadedMime.includes('png')?'.png':'.jpg');

  return `${roadName}_${yymm}_${detail}${suffix?'_'+suffix:''}${ext}`;

}


async function saveDocCopy(filename, propId, category) {

  if(!_lastUploadedBase64 || !_lastUploadedMime) return;

  const folderName = propFirstWord(propId); // e.g. "Maple" not "14 Maple Close"

  if(gAccessToken) {

    showSync('⟳ Uploading to Drive…');

    const link = await uploadFileToDrive(_lastUploadedBase64, _lastUploadedMime, filename, folderName, category||'Documents');

    if(link) return;

  }

  const byteChars = atob(_lastUploadedBase64);

  const bytes = new Uint8Array(byteChars.length);

  for(let i=0;i<byteChars.length;i++) bytes[i]=byteChars.charCodeAt(i);

  const blob = new Blob([bytes],{type:_lastUploadedMime});

  const a = document.createElement('a');

  a.href = URL.createObjectURL(blob);

  a.download = filename;

  a.click();

  URL.revokeObjectURL(a.href);

}

function openCertDoc(certId) {

  const c = state.certs.find(x=>x.id===certId);

  if(!c || !c.docData) return;

  try {

    const byteChars = atob(c.docData);

    const bytes = new Uint8Array(byteChars.length);

    for(let i=0;i<byteChars.length;i++) bytes[i]=byteChars.charCodeAt(i);

    const blob = new Blob([bytes], {type: c.docMime||'application/pdf'});

    const url = URL.createObjectURL(blob);

    window.open(url, '_blank');

    setTimeout(()=>URL.revokeObjectURL(url), 10000);

  } catch(e) {

    alert('Could not open document.');

  }

}

function npTickChanged() {

  const tickedExisting = state.owners.filter(o => document.getElementById('np-ck-'+o.id)?.checked);

  const tickedNewCount = [0,1,2,3].filter(i => document.getElementById('np-ck-new-'+i)?.checked).length;

  const totalTicked = tickedExisting.length + tickedNewCount;

  if(totalTicked > 0) {

    const each = +(100/totalTicked).toFixed(1);

    const last = +(100 - each*(totalTicked-1)).toFixed(1);

    let idx = 0;

    tickedExisting.forEach(o => {

      const el = document.getElementById('np-own-share-'+o.id);

      if(el) { el.value = idx===totalTicked-1 ? last : each; idx++; }

    });

    state.owners.filter(o => !document.getElementById('np-ck-'+o.id)?.checked).forEach(o => {

      const el = document.getElementById('np-own-share-'+o.id);

      if(el) el.value = '';

    });

    [0,1,2,3].forEach(i => {

      const el = document.getElementById('np-new-owner-share-'+i);

      if(el) {

        if(document.getElementById('np-ck-new-'+i)?.checked) {

          el.value = idx===totalTicked-1 ? last : each;

          idx++;

        } else {

          el.value = '';

        }

      }

    });

  } else {

    state.owners.forEach(o => { const el=document.getElementById('np-own-share-'+o.id); if(el) el.value=''; });

    [0,1,2,3].forEach(i => { const el=document.getElementById('np-new-owner-share-'+i); if(el) el.value=''; });

  }

  updateNewPropOwnerTotal();

}

function saveProp() {

  const s1 = window._newPropStep1;

  if(!s1 || !s1.name){ alert('Please enter a property name.'); return; }

  const propOwners = [];

  let totalShare = 0;

  state.owners.forEach(o => {

    const share = parseFloat(document.getElementById('np-own-share-'+o.id)?.value||0);

    if(share > 0) { propOwners.push({ownerId:o.id, share}); totalShare+=share; }

  });

  [0,1,2,3].forEach(i => {

    const nameEl = document.getElementById('np-new-owner-name-'+i);

    const shareEl = document.getElementById('np-new-owner-share-'+i);

    const name = nameEl?.value?.trim();

    const share = parseFloat(shareEl?.value||0);

    if(name) {

      let owner = state.owners.find(o=>o.name.toLowerCase()===name.toLowerCase());

      if(!owner) {

        owner = {id: uid(), name};

        state.owners.push(owner);

      }

      if(share > 0) { propOwners.push({ownerId:owner.id, share}); totalShare+=share; }

      else { propOwners.push({ownerId:owner.id, share:0}); }

    }

  });

  if(totalShare > 0 && Math.abs(totalShare-100) > 1) {

    alert(`Ownership shares add up to ${totalShare.toFixed(0)}% — they must total 100%. Please adjust.`);

    return;

  }

  if(propOwners.length > 0 && totalShare === 0) {

    const each = +(100/propOwners.length).toFixed(1);

    propOwners.forEach((o,i) => { o.share = i===propOwners.length-1 ? 100-(each*(propOwners.length-1)) : each; });

  }

  const myShare = propOwners.length > 0 ? (propOwners[0]?.share ?? 100) : 100;


  const idx = state.props.length % PROP_COLORS.length;

  state.props.push({

    id: uid(), name: s1.name, address: s1.addr,

    status: s1.status, color: PROP_COLORS[idx],

    ownershipShare: myShare,

    numOwners: propOwners.length || 1,

    owners: propOwners.length > 0 ? propOwners : [],

    tenant: s1.tenant||'—', leaseStart: s1.leaseStart||'—', leaseEnd:'—', deposit:'—',

    tenantEmail:'—', tenantPhone:'—',

    rent: parseFloat(s1.rent)||0,

    lastRentIncrease: '',

    lastRentAmount: 0,

    purchase: parseFloat(s1.purchase)||0,

    value: parseFloat(s1.value)||0,

    mortgage: parseFloat(s1.mortgage)||0,

    lender: s1.lender||'—',

    agent:'—', agentTel:'—', gasEngineer:'—', electrician:'—', solicitor:'—', insurance:'—',

    archived: false,

  });

  const newProp = state.props[state.props.length-1];

  if(newProp.rent > 0) {

    if(!state.rentHistory) state.rentHistory = [];

    state.rentHistory.push({

      id: uid(), propId: newProp.id,

      amount: newProp.rent,

      tenant: s1.tenant||'',

      startDate: new Date().toISOString().slice(0,10),

      endDate: null,

      notes: 'Initial rent (added with property)'

    });

    saveState('rentHistory');

  }


  saveState('props');

  saveState('owners');

  window._newPropStep1 = null;

  closeModal();

  populateOwnerSelect();

  renderProperties();

  renderDashboard();

}

function saveOwnership(propId) {

  const p = getProp(propId);

  if(!p) return;

  let total = 0;

  const owners = state.owners.map(o=>{

    const share = parseFloat(document.getElementById('own-'+o.id)?.value||0);

    total += share;

    return {ownerId:o.id,share};

  }).filter(x=>x.share>0);

  if(owners.length > 0 && total > 0 && Math.abs(total-100)>1) {

    alert('Total ownership must equal 100%. Current total: ' + total.toFixed(0) + '%');

    return;

  }

  if(owners.length === 0 || total === 0) {

    p.owners = [];

    p.ownershipShare = 100;

    p.numOwners = 1;

    saveState('props'); closeModal(); renderPropDetailContent(propId); renderAll();

  p.ownershipShare = myOwner?.share ?? 100;

  p.numOwners = owners.length;

}
